# POP PN Performance Report — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Python pipeline that reads a MoEngage push notification export, enriches it with 40+ derived columns (BU tags, time cuts, copy analysis, tonality DO/DON'T labels, funnel metrics, A/B detection, MOM/WOW deltas), and writes 7 analysis-ready tabs to a Google Sheet that powers a Looker Studio CMO dashboard.

**Architecture:** A single entry-point script (`run_report.py`) orchestrates a pipeline of focused enricher modules — each adding their own columns to a pandas DataFrame — then passes the enriched master table to summary builders which produce aggregated output tables, all written to Google Sheets via `gspread`. All enrichers are pure functions that take a DataFrame and return a DataFrame, making each one independently testable.

**Tech Stack:** Python 3.11+, pandas, gspread, gspread-dataframe, emoji, python-dotenv, pytest

```python
# src/loader.py
import pandas as pd
import os
from config import COL_ALL_SENT


def load_from_csv(path: str) -> pd.DataFrame:
    """Load MoEngage export from CSV. Excludes rows with 0 sent."""
    df = pd.read_csv(path, dtype=str)
    df[COL_ALL_SENT] = pd.to_numeric(df[COL_ALL_SENT], errors='coerce').fillna(0)
    return df[df[COL_ALL_SENT] > 0].reset_index(drop=True)


def load_lookup_from_csv(path: str) -> pd.DataFrame:
    """Load shop lookup table from CSV."""
    return pd.read_csv(path, dtype=str).fillna('')


def load_from_sheets(sheet_id: str, key_path: str) -> tuple[pd.DataFrame, pd.DataFrame]:
    """Load raw_input and shop_lookup tabs from Google Sheets."""
    import gspread
    from gspread_dataframe import get_as_dataframe

    gc = gspread.service_account(filename=key_path)
    sh = gc.open_by_key(sheet_id)

    raw_ws = sh.worksheet('raw_input')
    raw_df = get_as_dataframe(raw_ws, evaluate_formulas=True).dropna(how='all')
    raw_df[COL_ALL_SENT] = pd.to_numeric(raw_df[COL_ALL_SENT], errors='coerce').fillna(0)
    raw_df = raw_df[raw_df[COL_ALL_SENT] > 0].reset_index(drop=True)

    lookup_ws = sh.worksheet('shop_lookup')
    lookup_df = get_as_dataframe(lookup_ws, evaluate_formulas=True).dropna(how='all').fillna('')

    return raw_df, lookup_df
```

```python
# src/bu_tagger.py
import re
import pandas as pd
from config import (
    COL_TAG_POPCARD, COL_TAG_RUPAY, COL_TAG_UNCATEGORIZED, COL_TAG_SHOP,
    BU_NAMED_TAGS, BU_UNCATEGORIZED,
)


def _parse_tag_list(value: str) -> list[str]:
    """Extract string values from MoEngage's list-as-string: "['UPI']" → ['UPI']"""
    if not isinstance(value, str) or value.strip() in ('[]', '', 'nan'):
        return []
    return re.findall(r"'([^']+)'", value)


def _detect_bus(row: pd.Series) -> list[str]:
    """Return all BU names detected for a single row."""
    found = []
    # Named tag categories
    for bu_name, col in BU_NAMED_TAGS.items():
        if col in row.index and _parse_tag_list(str(row[col])):
            found.append(bu_name)
    # Uncategorized tag column
    if COL_TAG_UNCATEGORIZED in row.index:
        tags = _parse_tag_list(str(row[COL_TAG_UNCATEGORIZED]))
        for tag in tags:
            if tag in BU_UNCATEGORIZED:
                found.append(tag)
    return found if found else ['Unknown']


def tag_bu(df: pd.DataFrame) -> pd.DataFrame:
    """
    Add 'bu' and 'is_multi_bu' columns. Multi-BU rows are duplicated
    (one row per BU) with is_multi_bu=True.
    """
    rows = []
    for _, row in df.iterrows():
        bus = _detect_bus(row)
        is_multi = len(bus) > 1
        for bu in bus:
            new_row = row.copy()
            new_row['bu'] = bu
            new_row['is_multi_bu'] = is_multi
            rows.append(new_row)
    return pd.DataFrame(rows).reset_index(drop=True)
```

```python
# src/time_enricher.py
import pandas as pd
from config import (
    COL_SENT_TIME, TIME_SLOTS, PAYDAY_DAYS,
    PRE_JUNE_START, POST_JUNE_START, BRAND_ERA_PRE, BRAND_ERA_POST,
)


def _time_slot(hour: int) -> str:
    for name, start, end in TIME_SLOTS:
        if start <= hour < end:
            return name
    return 'Other'


def enrich_time(df: pd.DataFrame) -> pd.DataFrame:
    df = df.copy()
    df['_dt'] = pd.to_datetime(df[COL_SENT_TIME], errors='coerce')

    df['sent_date']         = df['_dt'].dt.date
    df['sent_hour']         = df['_dt'].dt.hour
    df['sent_day_of_week']  = df['_dt'].dt.strftime('%A')
    df['sent_week']         = df['_dt'].dt.isocalendar().week.astype(int)
    df['sent_month']        = df['_dt'].dt.to_period('M').astype(str)
    df['is_weekend']        = df['_dt'].dt.dayofweek >= 5
    df['time_slot_bucket']  = df['sent_hour'].apply(_time_slot)
    df['day_of_month_bucket'] = df['_dt'].dt.day.apply(
        lambda d: 'Payday Week' if d in PAYDAY_DAYS else 'Rest of Month'
    )
    df['brand_guidelines_era'] = df['sent_date'].apply(
        lambda d: BRAND_ERA_PRE if d is not None and d < POST_JUNE_START else BRAND_ERA_POST
    )

    # days_since_last_pn_bu — per BU, sorted by date
    df = df.sort_values(['bu', '_dt']).reset_index(drop=True)
    df['days_since_last_pn_bu'] = (
        df.groupby('bu')['_dt']
        .diff()
        .dt.total_seconds()
        .div(86400)
        .round(1)
    )

    # same_day_pn_count — count of PNs sent on same calendar date (all BUs)
    day_counts = df.groupby('sent_date')['sent_date'].transform('count')
    df['same_day_pn_count'] = day_counts

    # pn_sequence_position — rank within the day by sent_hour
    df['pn_sequence_position'] = (
        df.sort_values('_dt')
        .groupby('sent_date')
        .cumcount() + 1
    )

    df.drop(columns=['_dt'], inplace=True)
    return df
```

```python
# src/copy_analyser.py
import re
import emoji as emoji_lib
import pandas as pd
from config import (
    COL_ANDROID_TITLE, COL_ANDROID_BODY, COL_RICH_IMAGE,
    ACTION_VERBS, FOMO_WORDS, CULTURAL_REFS,
    FORCED_GENZ_WORDS, CORPORATE_JARGON_WORDS,
)

_NUMBER_RE = re.compile(r'₹\s*\d+|\d+\s*POPcoins|\d+\s*%', re.IGNORECASE)


def _extract_emojis(text: str) -> list[str]:
    return [ch for ch in text if ch in emoji_lib.EMOJI_DATA]


def _emoji_position(text: str, emojis: list[str]) -> str:
    if not emojis:
        return 'None'
    stripped = text.strip()
    if stripped and stripped[0] in emoji_lib.EMOJI_DATA:
        return 'Start'
    if stripped and stripped[-1] in emoji_lib.EMOJI_DATA:
        return 'End'
    return 'Middle'


def _word_count(text: str) -> int:
    return len(str(text).split())


def _contains_any(text: str, phrases: list[str]) -> bool:
    lower = text.lower()
    return any(p.lower() in lower for p in phrases)


def _analyse_row(row: pd.Series) -> dict:
    title = str(row.get(COL_ANDROID_TITLE, '') or '')
    body  = str(row.get(COL_ANDROID_BODY, '') or '')
    image = str(row.get(COL_RICH_IMAGE, '') or '')
    full  = title + ' ' + body

    emojis          = _extract_emojis(title)
    emoji_count     = len(emojis)
    title_words     = _word_count(title)
    body_words      = _word_count(body)

    return {
        'has_emoji':             emoji_count > 0,
        'emoji_count':           emoji_count,
        'emoji_count_bucket':    '0' if emoji_count == 0 else ('1' if emoji_count == 1 else '2+'),
        'emoji_position':        _emoji_position(title, emojis),
        'title_char_length':     len(title),
        'title_word_count':      title_words,
        'title_length_bucket':   ('Short' if title_words <= 5 else ('Medium' if title_words <= 9 else 'Long')),
        'body_word_count':       body_words,
        'body_length_bucket':    ('Short' if body_words < 10 else ('Medium' if body_words <= 20 else 'Long')),
        'has_personalisation':   any(w in full.lower().split() for w in ('you', 'your')),
        'has_specific_number':   bool(_NUMBER_RE.search(full)),
        'has_action_verb':       _contains_any(title, ACTION_VERBS),
        'has_exclamation':       '!' in title,
        'has_question_mark':     '?' in title,
        'has_fomo_signal':       _contains_any(full, FOMO_WORDS),
        'has_cultural_reference': _contains_any(full, CULTURAL_REFS),
        'has_rich_media':        bool(image.strip()),
        'is_forced_genz':        _contains_any(full, FORCED_GENZ_WORDS),
        'is_corporate_jargon':   _contains_any(full, CORPORATE_JARGON_WORDS),
    }


def analyse_copy(df: pd.DataFrame) -> pd.DataFrame:
    df = df.copy()
    flags = df.apply(_analyse_row, axis=1, result_type='expand')
    return pd.concat([df, flags], axis=1)
```

```python
# src/tonality_classifier.py
import re
import pandas as pd
from config import (
    COL_ANDROID_TITLE, COL_ANDROID_BODY,
    LECTURE_Y_PHRASES, CLICHE_PHRASES, CONDESCENDING_PHRASES, VAGUE_PHRASES,
)

_WITTY_PATTERNS = re.compile(
    r'"[^"]+"|\?.*\!|era starts|gotchu|we see you|plot twist|ngl|tbh',
    re.IGNORECASE,
)
_FRIENDLY_PATTERNS = re.compile(
    r'\bwe\b|\bgotchu\b|\bwe got\b|\byou deserve\b|\bfor you\b',
    re.IGNORECASE,
)
_HELPFUL_PATTERNS = re.compile(
    r'bill|spend|save time|pay faster|manage|track|reminder|due',
    re.IGNORECASE,
)

def _contains(text: str, phrases: list[str]) -> bool:
    lower = text.lower()
    return any(p.lower() in lower for p in phrases)


def _classify_row(row: pd.Series) -> dict:
    title = str(row.get(COL_ANDROID_TITLE, '') or '')
    body  = str(row.get(COL_ANDROID_BODY, '') or '')
    full  = (title + ' ' + body).strip()
    body_words = len(body.split())

    # ── DON'T checks (priority order) ────────────────────────────────────────
    if row.get('is_forced_genz', False):
        tone = "DON'T: Forced Gen Z"
    elif row.get('is_corporate_jargon', False):
        tone = "DON'T: Corporate Jargon"
    elif _contains(full, CONDESCENDING_PHRASES):
        tone = "DON'T: Condescending"
    elif _contains(full, LECTURE_Y_PHRASES) or body_words > 30:
        tone = "DON'T: Lecture-y"
    elif _contains(full, CLICHE_PHRASES):
        tone = "DON'T: Cliche"
    elif _contains(full, VAGUE_PHRASES) and not row.get('has_specific_number', False):
        tone = "DON'T: Vague"

    # ── DO checks (priority order) ────────────────────────────────────────────
    elif row.get('has_specific_number', False):
        tone = 'DO: Smart — Value-aware'
    elif bool(_WITTY_PATTERNS.search(full)):
        tone = 'DO: Smart — Unique'
    elif row.get('has_cultural_reference', False):
        tone = 'DO: Relatable — Youthful'
    elif bool(_FRIENDLY_PATTERNS.search(full)) or row.get('has_personalisation', False):
        tone = 'DO: Relatable — Friendly'
    elif bool(_HELPFUL_PATTERNS.search(full)):
        tone = 'DO: Relatable — Helpful'
    else:
        tone = 'DO: Smart — Simple'

    parent  = tone.split(':')[0].strip()
    subtype = tone.split(':', 1)[1].strip() if ':' in tone else tone

    return {
        'tonality':         tone,
        'tonality_parent':  parent,
        'tonality_subtype': subtype,
        'brand_compliant':  parent == 'DO',
    }


def classify_tonality(df: pd.DataFrame) -> pd.DataFrame:
    df = df.copy()
    classified = df.apply(_classify_row, axis=1, result_type='expand')
    return pd.concat([df, classified], axis=1)
```

```python
import pandas as pd
from config import (
    COL_ALL_SENT, COL_ALL_IMPRESSIONS, COL_ALL_CLICKS,
    COL_ALL_AFTER_FC, COL_ALL_INSTALLED, GOAL_CONVERTED_COLS,
)

def _safe_div(num: float, den: float) -> float:
    return round(num / den, 6) if den and den != 0 else 0.0

def _primary_conversions(row: pd.Series) -> float:
    for col in GOAL_CONVERTED_COLS:
        val = pd.to_numeric(row.get(col, 0), errors='coerce') or 0
        if val > 0:
            return val
    return 0.0

def add_funnel_metrics(df: pd.DataFrame) -> pd.DataFrame:
    df = df.copy()
    for col in [COL_ALL_SENT, COL_ALL_IMPRESSIONS, COL_ALL_CLICKS,
                COL_ALL_AFTER_FC, COL_ALL_INSTALLED] + GOAL_CONVERTED_COLS:
        if col in df.columns:
            df[col] = pd.to_numeric(df[col], errors='coerce').fillna(0)

    df['primary_conversions']      = df.apply(_primary_conversions, axis=1)
    df['reachability_rate']        = df.apply(lambda r: _safe_div(r[COL_ALL_AFTER_FC], r[COL_ALL_INSTALLED]), axis=1)
    df['fc_hit_rate']              = df.apply(lambda r: 1 - _safe_div(r[COL_ALL_AFTER_FC], r[COL_ALL_INSTALLED]), axis=1)
    df['sent_to_impression_rate']  = df.apply(lambda r: _safe_div(r[COL_ALL_IMPRESSIONS], r[COL_ALL_SENT]), axis=1)
    df['impression_to_click_rate'] = df.apply(lambda r: _safe_div(r[COL_ALL_CLICKS], r[COL_ALL_IMPRESSIONS]), axis=1)
    df['click_to_convert_rate']    = df.apply(lambda r: _safe_div(r['primary_conversions'], r[COL_ALL_CLICKS]), axis=1)
    df['end_to_end_funnel_rate']   = df.apply(lambda r: _safe_div(r['primary_conversions'], r[COL_ALL_SENT]), axis=1)
    return df
```

```python
from config import COL_CAMPAIGN_ID, COL_VARIATION, COL_ALL_CTR, COL_GOAL1_CONVERTED, COL_ALL_CLICKS

def detect_ab(df: pd.DataFrame) -> pd.DataFrame:
    df = df.copy()
    for col in [COL_ALL_CTR, COL_GOAL1_CONVERTED, COL_ALL_CLICKS]:
        if col in df.columns:
            df[col] = pd.to_numeric(df[col], errors='coerce').fillna(0)

    variation_counts = df.groupby(COL_CAMPAIGN_ID)[COL_VARIATION].transform('nunique')
    df['is_ab_test'] = variation_counts > 1

    df['ab_winner']   = False
    df['ab_lift_ctr'] = 0.0

    for camp_id, group in df[df['is_ab_test']].groupby(COL_CAMPAIGN_ID):
        max_ctr = group[COL_ALL_CTR].max()
        winner_mask = group[COL_ALL_CTR] == max_ctr
        min_ctr = group[COL_ALL_CTR].min()
        lift = round(max_ctr - min_ctr, 4)

        df.loc[group[winner_mask].index, 'ab_winner']   = True
        df.loc[group.index, 'ab_lift_ctr'] = lift

    return df
```

```python
# src/shop_enricher.py
import pandas as pd
from config import COL_CAMPAIGN_ID

def enrich_shop(df: pd.DataFrame, lookup: pd.DataFrame) -> pd.DataFrame:
    df = df.copy()
    df['shop_category'] = ''
    df['shop_brand']    = ''
    df['shop_product']  = ''

    if lookup.empty:
        return df

    shop_mask = df['bu'] == 'Shop'
    if not shop_mask.any():
        return df

    merged = df[shop_mask].merge(
        lookup[['campaign_id', 'shop_category', 'shop_brand', 'shop_product']],
        left_on=COL_CAMPAIGN_ID, right_on='campaign_id', how='left', suffixes=('', '_lkp')
    )
    for col in ['shop_category', 'shop_brand', 'shop_product']:
        lkp_col = col + '_lkp' if col + '_lkp' in merged.columns else col
        df.loc[shop_mask, col] = merged[lkp_col].fillna('').values

    return df
```

```python
import pandas as pd
```

```python
               COL_ALL_CTR, 'primary_conversions', 'end_to_end_funnel_rate',
               'reachability_rate', COL_ALL_FCM_RATE]

def _mom_wow_deltas(df: pd.DataFrame, group_cols: list, period_col: str) -> pd.DataFrame:
    df = df.sort_values(group_cols + [period_col])
    for col in METRIC_COLS:
        if col in df.columns:
            df[f'mom_{col}_delta']     = df.groupby(group_cols)[col].diff()
            df[f'mom_{col}_delta_pct'] = df.groupby(group_cols)[col].pct_change().mul(100).round(2)
    return df

def build_summary_overall(master: pd.DataFrame) -> pd.DataFrame:
    for col in METRIC_COLS:
        if col in master.columns:
            master[col] = pd.to_numeric(master[col], errors='coerce').fillna(0)

        **{col: (col, 'sum' if col in [COL_ALL_SENT, COL_ALL_IMPRESSIONS,
                                        COL_ALL_CLICKS, 'primary_conversions']
                 else 'mean') for col in METRIC_COLS if col in master.columns},
        campaign_count=('Campaign ID', 'nunique'),
    ).reset_index().rename(columns={'sent_month': 'period_label'})

    monthly = _mom_wow_deltas(monthly, [], 'period_label')
    return monthly
```

```python
# src/summary_bu.py
import pandas as pd
from config import COL_ALL_SENT, COL_ALL_IMPRESSIONS, COL_ALL_CLICKS, COL_ALL_CTR, COL_ALL_FCM_RATE

METRIC_COLS = [COL_ALL_SENT, COL_ALL_IMPRESSIONS, COL_ALL_CLICKS,
               COL_ALL_CTR, 'primary_conversions', 'end_to_end_funnel_rate',
               'reachability_rate', COL_ALL_FCM_RATE]

def _agg_by_period(master: pd.DataFrame, period_col: str) -> pd.DataFrame:
    agg_dict = {
        col: (col, 'sum' if col in [COL_ALL_SENT, COL_ALL_IMPRESSIONS,
                                     COL_ALL_CLICKS, 'primary_conversions']
              else 'mean')
        for col in METRIC_COLS if col in master.columns
    }
    agg_dict['campaign_count'] = ('Campaign ID', 'nunique')
    agg_dict['ab_test_count']  = ('is_ab_test', 'sum')

    return (
        master.groupby(['bu', period_col])
        .agg(**agg_dict)
        .reset_index()
        .rename(columns={period_col: 'period_label'})
        .sort_values(['bu', 'period_label'])
    )


def build_summary_bu(master: pd.DataFrame) -> pd.DataFrame:
    for col in METRIC_COLS:
        if col in master.columns:
            master[col] = pd.to_numeric(master[col], errors='coerce').fillna(0)

    # Monthly view (MOM deltas)
    monthly_bu = _agg_by_period(master, 'sent_month')
    monthly_bu['period_type'] = 'Monthly'
    for col in METRIC_COLS:
        if col in monthly_bu.columns:
            monthly_bu[f'mom_{col}_delta'] = monthly_bu.groupby('bu')[col].diff()
    monthly_bu['mom_ctr_delta_pct'] = (
        monthly_bu.groupby('bu')[COL_ALL_CTR].pct_change().mul(100).round(2)
    )

    # Weekly view (WOW deltas) — cast week to str for readability
    master['sent_week_label'] = master['sent_month'].str[:7] + '-W' + master['sent_week'].astype(str).str.zfill(2)
    weekly_bu = _agg_by_period(master, 'sent_week_label')
    weekly_bu['period_type'] = 'Weekly'
    for col in METRIC_COLS:
        if col in weekly_bu.columns:
            weekly_bu[f'wow_{col}_delta'] = weekly_bu.groupby('bu')[col].diff()
    weekly_bu['wow_ctr_delta_pct'] = (
        weekly_bu.groupby('bu')[COL_ALL_CTR].pct_change().mul(100).round(2)
    )

    return pd.concat([monthly_bu, weekly_bu], ignore_index=True)
```

```python
def build_top_bottom(master: pd.DataFrame) -> pd.DataFrame:

    df[COL_ALL_SENT] = pd.to_numeric(df[COL_ALL_SENT], errors='coerce').fillna(0)

        bottom['rank_type'] = 'Bottom'
```

```python
    df[COL_ALL_SENT] = pd.to_numeric(df[COL_ALL_SENT], errors='coerce').fillna(0)
    df['primary_conversions'] = pd.to_numeric(df.get('primary_conversions', 0), errors='coerce').fillna(0)

    # By era + month
    by_era_month = (
        df.groupby(['brand_guidelines_era', 'sent_month'])
        .agg(
            campaign_count=('Campaign ID', 'nunique'),
            compliant_count=('brand_compliant', 'sum'),
            avg_ctr=(COL_ALL_CTR, 'mean'),
            avg_conversions=('primary_conversions', 'mean'),
            forced_genz_count=('is_forced_genz', 'sum'),
            corporate_jargon_count=('is_corporate_jargon', 'sum'),
        )
        .reset_index()
    )
    by_era_month['compliance_rate'] = (
        by_era_month['compliant_count'] / by_era_month['campaign_count']
    ).round(4)

    # By era + bu
    by_era_bu = (
        df.groupby(['brand_guidelines_era', 'bu'])
        .agg(
            campaign_count=('Campaign ID', 'nunique'),
            compliant_count=('brand_compliant', 'sum'),
            avg_ctr=(COL_ALL_CTR, 'mean'),
        )
        .reset_index()
    )
    by_era_bu['compliance_rate'] = (
        by_era_bu['compliant_count'] / by_era_bu['campaign_count']
    ).round(4)

    # Compliant vs non-compliant CTR
    by_compliance = (
        df.groupby(['brand_guidelines_era', 'brand_compliant'])
        .agg(avg_ctr=(COL_ALL_CTR, 'mean'), campaign_count=('Campaign ID', 'nunique'))
        .reset_index()
    )

    # Tag and concat all three into one DataFrame with a 'table_type' column
    by_era_month['table_type']   = 'era_month'
    by_era_bu['table_type']      = 'era_bu'
    by_compliance['table_type']  = 'compliance_comparison'

    return pd.concat([by_era_month, by_era_bu, by_compliance], ignore_index=True)
```

```python
# src/sheets_writer.py
import pandas as pd
import gspread
from gspread_dataframe import set_with_dataframe

OUTPUT_TABS = {
    'master_enriched':        None,
    'summary_overall':        None,
    'summary_by_bu':          None,
    'top_bottom_campaigns':   None,
    'copy_analysis':          None,
    'ab_test_results':        None,
    'brand_guidelines_impact': None,
}

def _ensure_worksheet(sh: gspread.Spreadsheet, name: str) -> gspread.Worksheet:
    """Get worksheet by name, create it if it doesn't exist."""
    try:
        return sh.worksheet(name)
    except gspread.exceptions.WorksheetNotFound:
        return sh.add_worksheet(title=name, rows=5000, cols=100)


def write_all_tabs(
    sheet_id: str,
    key_path: str,
    master: pd.DataFrame,
    summary_overall: pd.DataFrame,
    summary_bu: pd.DataFrame,
    top_bottom: pd.DataFrame,
    copy_analysis: pd.DataFrame,
    ab_results: pd.DataFrame,
    brand_impact: pd.DataFrame,
) -> None:
    """Write all 7 output tabs to Google Sheets, overwriting existing data."""
    gc = gspread.service_account(filename=key_path)
    sh = gc.open_by_key(sheet_id)

    tabs = {
        'master_enriched':         master,
        'summary_overall':         summary_overall,
        'summary_by_bu':           summary_bu,
        'top_bottom_campaigns':    top_bottom,
        'copy_analysis':           copy_analysis,
        'ab_test_results':         ab_results,
        'brand_guidelines_impact': brand_impact,
    }

    for tab_name, df in tabs.items():
        ws = _ensure_worksheet(sh, tab_name)
        ws.clear()
        if df is not None and not df.empty:
            # Convert all columns to string to avoid serialisation issues with dates/bools
            df_out = df.copy().astype(str)
            set_with_dataframe(ws, df_out, include_index=False, resize=True)
            print(f'  ✓ {tab_name}: {len(df_out)} rows written')
        else:
            print(f'  ✗ {tab_name}: empty, skipped')
```
